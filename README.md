## Setup Git and GitHub Global Configuration ##
$ git config --global user.email "gautamthakur1983@gmail.com"
$ git config --global user.name "Gautam Thakur"
$ git config --global list
$ git config --list

$ git status
$ git init
$ git add .
$ git commit -m "Commit Message"
